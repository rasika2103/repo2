7. Paste This Full Code in index.html

<!DOCTYPE html>
<html>
<head>
  <title>Firebase Authentication Practical</title>
  <style>
    body {
      font-family: Arial;
      background: #f4f6f8;
      text-align: center;
      padding-top: 50px;
    }

    .box {
      background: white;
      width: 400px;
      margin: auto;
      padding: 25px;
      border-radius: 10px;
      box-shadow: 0 0 10px gray;
    }

    input {
      width: 90%;
      padding: 12px;
      margin: 10px;
      font-size: 16px;
    }

    button {
      padding: 12px 25px;
      margin: 10px;
      background: #1a73e8;
      color: white;
      border: none;
      cursor: pointer;
      border-radius: 5px;
      font-size: 16px;
    }

    button:hover {
      background: #0b57d0;
    }

    #message {
      margin-top: 20px;
      font-weight: bold;
      color: green;
    }
  </style>
</head>

<body>

  <div class="box">
    <h1>Firebase Authentication</h1>
    <h3>Register / Login User</h3>

    <input type="email" id="email" placeholder="Enter Email">
    <br>

    <input type="password" id="password" placeholder="Enter Password">
    <br>

    <button onclick="registerUser()">Register</button>
    <button onclick="loginUser()">Login</button>

    <p id="message"></p>
  </div>

  <script type="module">

    import { initializeApp } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-app.js";

    import {
      getAuth,
      createUserWithEmailAndPassword,
      signInWithEmailAndPassword
    } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-auth.js";

    import {
      getFirestore,
      doc,
      setDoc
    } from "https://www.gstatic.com/firebasejs/10.12.2/firebase-firestore.js";

    const firebaseConfig = {
      apiKey: "PASTE_YOUR_API_KEY",
      authDomain: "PASTE_YOUR_AUTH_DOMAIN",
      projectId: "PASTE_YOUR_PROJECT_ID",
      storageBucket: "PASTE_YOUR_STORAGE_BUCKET",
      messagingSenderId: "PASTE_YOUR_MESSAGING_SENDER_ID",
      appId: "PASTE_YOUR_APP_ID"
    };

    const app = initializeApp(firebaseConfig);
    const auth = getAuth(app);
    const db = getFirestore(app);

    window.registerUser = async function () {
      const email = document.getElementById("email").value;
      const password = document.getElementById("password").value;

      try {
        const userCredential = await createUserWithEmailAndPassword(auth, email, password);

        await setDoc(doc(db, "users", userCredential.user.uid), {
          uid: userCredential.user.uid,
          email: email,
          createdAt: new Date().toString(),
          status: "Registered Successfully"
        });

        document.getElementById("message").innerText =
          "User registered and stored successfully";

      } catch (error) {
        document.getElementById("message").innerText = error.message;
      }
    }

    window.loginUser = async function () {
      const email = document.getElementById("email").value;
      const password = document.getElementById("password").value;

      try {
        const userCredential = await signInWithEmailAndPassword(auth, email, password);

        await setDoc(doc(db, "users", userCredential.user.uid), {
          uid: userCredential.user.uid,
          email: email,
          lastLogin: new Date().toString(),
          status: "Login Verified"
        }, { merge: true });

        document.getElementById("message").innerText =
          "Login successful. User verified and updated in Firestore";

      } catch (error) {
        document.getElementById("message").innerText = error.message;
      }
    }

  </script>

</body>
</html>
