from google.appengine.ext import webapp
from google.appengine.ext.webapp import util


class MainHandler(webapp.RequestHandler):
    def get(self):
        self.response.headers['Content-Type'] = 'text/html'
        self.response.out.write("""
        <html>
        <head>
            <title>GAE Practical</title>
        </head>

        <body style="font-family:Arial; text-align:center; padding-top:100px; background:#eef2ff;">

            <div style="background:white;
                        width:60%;
                        margin:auto;
                        padding:40px;
                        border-radius:15px;
                        box-shadow:0px 0px 10px gray;">

                <h1 style="color:#1a73e8;">
                    Google App Engine Practical
                </h1>

                <p>
                    Web application launched successfully using GAE Launcher.
                </p>

                <p>
                    Practical Assignment 3
                </p>

            </div>

        </body>
        </html>
        """)


def main():
    application = webapp.WSGIApplication([('/', MainHandler)],
                                         debug=True)
    util.run_wsgi_app(application)


if __name__ == '__main__':
    main()
